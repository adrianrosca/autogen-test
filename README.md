# get started
- pnpm i
- pnpm generate (F5)
- pnpm start

# clean
- pnpm clean

# create new repo from existing code
- remove .git folder
- git init
- git add .
- git commit -m "Initial commit"
- [create repo on github]
- git remote add origin https://github.com/yourusername/your-repo-name.git
- git push -u origin master
